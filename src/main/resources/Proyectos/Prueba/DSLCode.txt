FILE(#country, country.csv)
FILE(#country_population, country_population.csv)
FILE(#region, region.csv)
PREFIX(Prueba, http://www.example.com/Prueba#)
SUBJECT(#country.countryCode,Prueba:country)
PREDICATE-OBJECT(PREDICATE(Prueba:countryCode),OBJECT(Prueba:#country.countryCode))
PREDICATE-OBJECT(PREDICATE(Prueba:name),OBJECT(Prueba:#country.name))
PREDICATE-OBJECT(PREDICATE(Prueba:continent),OBJECT(Prueba:#country.continent))
PREDICATE-OBJECT(PREDICATE(Prueba:capital),OBJECT(CONTAINER(SEQ,#country.capital)))
PREDICATE-OBJECT(PREDICATE(Prueba:area),OBJECT(Prueba:#country.area))
SUBJECT(#country_population.population,Prueba:country_population)
PREDICATE-OBJECT(PREDICATE(Prueba:countryCode),OBJECT(Prueba:#country_population.countryCode))
PREDICATE-OBJECT(PREDICATE(Prueba:population),OBJECT(Prueba:#country_population.population))
PREDICATE-OBJECT(PREDICATE(Prueba:period),OBJECT(#country_population.period,VALUES(0,1),ENUMS(l,h)))
SUBJECT(#region.regionCode,Prueba:region)
PREDICATE-OBJECT(PREDICATE(Prueba:regionCode),OBJECT(Prueba:#region.regionCode))
PREDICATE-OBJECT(PREDICATE(Prueba:name),OBJECT(Prueba:#region.name))
PREDICATE-OBJECT(PREDICATE(Prueba:countryCode),OBJECT(Prueba:#region.countryCode))
PREDICATE-OBJECT(PREDICATE(Prueba:area),OBJECT(CONTAINER(BAG,#region.area)))
